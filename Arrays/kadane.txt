class Solution{
    public:
    // arr: input array
    // n: size of array
    //Function to find the sum of contiguous subarray with maximum sum.
    int maxSubarraySum(int arr[], int n){
        
        // Your code here
        int sum=0,temp=0;
        
        int i;
        for(i=0;i<n;i++)
        {
            temp+=arr[i];
            if(temp>sum)
            {
                sum=temp;
            }
            if(temp<0)
            {
                temp=0;
            }
        }
        return sum;
    }
};
